**Current behavior:** Always connects to the rightmost node regardless of whether it already has an outgoing edge.

**New behavior per CONTEXT.md:** Connect to the nearest node without an outgoing edge. If none exists, don't auto-connect.

```typescript
function handleNodeDropped(nodeType: WorkflowNodeType, position: { x: number; y: number }) {
  const id = crypto.randomUUID();
  const label = DEFAULT_LABELS[nodeType];
  const config = buildDefaultConfig(nodeType);
  const outputs = deriveOutputs(id, config as NodeConfig);

  const newNode: WFNode = {
    id,
    type: nodeType,
    position,
    data: { nodeType, label, config, outputs },
  };

  // Find nodes that have no outgoing edge
  const nodesWithOutgoing = new Set(edges.map((e) => e.source));
  const candidates = nodes.filter((n) => !nodesWithOutgoing.has(n.id));

  if (candidates.length > 0) {
    // Connect to the nearest candidate (by Euclidean distance)
    const nearest = candidates.reduce((prev, curr) => {
      const distPrev = Math.hypot(
        (prev.position?.x ?? 0) - position.x,
        (prev.position?.y ?? 0) - position.y
      );
      const distCurr = Math.hypot(
        (curr.position?.x ?? 0) - position.x,
        (curr.position?.y ?? 0) - position.y
      );
      return distCurr < distPrev ? curr : prev;
    });

    const newEdge: WFEdge = {
      id: `e-${nearest.id}-${id}`,
      source: nearest.id,
      target: id,
      type: "dataflow",
    };
    setNodes([...nodes, newNode]);
    setEdges([...edges, newEdge]);
  } else {
    // No candidate without outgoing — just add node without connecting
    setNodes([...nodes, newNode]);
  }
}
```

Also import `NodeConfig` from `@intelliflow/shared` if not already imported (needed for the `as NodeConfig` cast in deriveOutputs call).
