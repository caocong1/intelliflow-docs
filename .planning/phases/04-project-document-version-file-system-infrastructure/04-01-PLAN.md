```typescript
// Project types
export type ProjectRole = "owner" | "participant";
export type DocumentVisibility = "self" | "project" | "specific";
export type DocumentStatus = "draft" | "in_progress" | "completed";

export interface Project extends BaseEntity {
  name: string;
  description: string | null;
  department: string | null;
  createdBy: string;
  isDeleted: boolean;
  deletedAt: string | null;
}

export interface ProjectMember {
  id: string;
  projectId: string;
  userId: string;
  role: ProjectRole;
  joinedAt: string;
  // Joined fields (optional, for list views)
  displayName?: string;
  username?: string;
}

export interface ProjectListItem extends Project {
  memberCount: number;
  userRole: ProjectRole | null; // null if user is not a member (admin viewing "all")
}

export interface Document extends BaseEntity {
  projectId: string;
  workflowId: string;
  title: string;
  description: string | null;
  status: DocumentStatus;
  visibility: DocumentVisibility;
  createdBy: string;
  isDeleted: boolean;
  deletedAt: string | null;
  isArchived: boolean;
  // Joined fields (optional)
  creatorName?: string;
  workflowName?: string;
}

export interface DocumentVersion {
  id: string;
  documentId: string;
  versionNumber: number;
  nodeId: string;
  nodeLabel: string;
  snapshotData: Record<string, unknown>;
  createdBy: string;
  createdAt: string;
  // Joined fields
  creatorName?: string;
}

export interface DocumentFile {
  id: string;
  documentId: string;
  category: "upload" | "export";
  originalName: string;
  storagePath: string;
  mimeType: string | null;
  fileSize: number | null;
  createdBy: string;
  createdAt: string;
}

export interface VersionDiffLine {
  type: "added" | "removed" | "unchanged";
  content: string;
  oldLineNumber?: number;
  newLineNumber?: number;
}

export interface VersionDiffResult {
  versionA: DocumentVersion;
  versionB: DocumentVersion;
  diffs: Record<string, VersionDiffLine[]>; // keyed by content field name
}
```

**New tables (following existing pattern: uuid PK, timestamps, references):**

1. `projects` table: id, name (varchar 200), description (varchar 1000 nullable), department (varchar 100 nullable), createdBy (uuid ref users.id), isDeleted (boolean default false), deletedAt (timestamp nullable), createdAt, updatedAt.

2. `projectMembers` table: id, projectId (uuid ref projects.id), userId (uuid ref users.id), role (projectRoleEnum default "participant"), joinedAt (timestamp defaultNow).

3. `documents` table: id, projectId (uuid ref projects.id), workflowId (uuid ref workflows.id), title (varchar 300), description (varchar 1000 nullable), status (documentStatusEnum default "draft"), visibility (documentVisibilityEnum default "project"), createdBy (uuid ref users.id), isDeleted (boolean default false), deletedAt (timestamp nullable), isArchived (boolean default false), createdAt, updatedAt.

4. `documentVisibilityMembers` table: id, documentId (uuid ref documents.id), userId (uuid ref users.id).

5. `documentVersions` table: id, documentId (uuid ref documents.id), versionNumber (integer), nodeId (varchar 100), nodeLabel (varchar 200), snapshotData (jsonb), createdBy (uuid ref users.id), createdAt (timestamp).

6. `documentFiles` table: id, documentId (uuid ref documents.id), category (varchar 20 — 'upload' or 'export'), originalName (varchar 500), storagePath (varchar 1000), mimeType (varchar 100 nullable), fileSize (integer nullable), createdBy (uuid ref users.id), createdAt (timestamp).
