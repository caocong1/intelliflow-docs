   c. Add rollback handler function that computes the target from workflowNodes config:
   ```typescript
   async function handleBlockedRollback(blockedNodeId: string) {
     const s = state();
     if (!s) return;

     // Look up the workflow node config to find executionRule.conditions
     const wfNode = s.workflowNodes.find((wn) => wn.id === blockedNodeId);
     const executionRule = (wfNode?.config as any)?.executionRule as NodeExecutionRule | undefined;

     if (!executionRule?.conditions?.length) return;

     // Collect all sourceRef.nodeId values from conditions
     const sourceNodeIds = executionRule.conditions.map(c => c.sourceRef.nodeId);

     // Find the earliest source node by stepOrder among current node executions
     const sourceExecs = s.nodes
       .filter(n => sourceNodeIds.includes(n.nodeId))
       .sort((a, b) => a.stepOrder - b.stepOrder);

     const targetNodeId = sourceExecs[0]?.nodeId;
     if (!targetNodeId) return;

     // Call rollback API
     const token = localStorage.getItem("token");
     const res = await fetch(`/api/runtime/${params.documentId}/rollback`, {
       method: "POST",
       headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` },
       body: JSON.stringify({ targetNodeId }),
     });
     if (res.ok) {
       refetch(); // refresh runtime state
     }
   }
   ```
