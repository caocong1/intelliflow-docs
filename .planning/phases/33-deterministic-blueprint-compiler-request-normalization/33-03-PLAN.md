const SENTINEL_MODEL_IDS = [SENTINEL_MODEL_ID];
const SENTINEL_LOCAL_MODEL_ID = "__pending_local_model__";
const DESENSITIZE_PLACEHOLDER_CATEGORIES = [
  { name: "default", description: "Phase 34 will populate real categories" },
];

type StageLookupEntry = { nodeId: string; namedOutputIds: string[] };

export function compileBlueprint(blueprint: FinalBlueprint): CompiledWorkflowDraft {
  assertBlueprintCompilable(blueprint);

  const nodes: WorkflowNodeDef[] = [];

  // Pre-sanitize + dedupe all stage ids across pre + post so that lookups can
  // use the emitted node ids consistently.
  const allStages: StageDraft[] = [
    ...blueprint.preRestoreStages,
    ...blueprint.postRestoreStages,
  ];
  const rawStageIds = allStages.map((s, idx) =>
    sanitizeMachineKey(s.id ?? "", `stage_${idx}`),
  );
  const dedupedStageIds = dedupeMachineKeys(rawStageIds);
  const preCount = blueprint.preRestoreStages.length;
  const postCount = blueprint.postRestoreStages.length;

  // 1. node_input
  nodes.push(makeInputNode(blueprint, nodes.length));

  // 2. node_desens (optional, D-14 placeholders so validator passes)
  if (blueprint.useDesensitize) {
    nodes.push(makeDesensNode(nodes.length));
  }

  // 3. Emit pre-restore stages — build lookup map as we go
  const emittedByOriginalIdPreRestore = new Map<string, StageLookupEntry>();
  for (let i = 0; i < preCount; i += 1) {
    const stage = allStages[i];
    const ensuredStage: StageDraft = { ...stage, id: dedupedStageIds[i] };
    const stageNode = compileStageNode(ensuredStage, nodes.length, SENTINEL_MODEL_IDS);
    nodes.push(stageNode);
    emittedByOriginalIdPreRestore.set(stage.id, {
      nodeId: stageNode.id,
      namedOutputIds: stage.namedOutputs.map((o) => o.id),
    });
  }

  // 4. node_restore (optional) — ONLY references pre-restore stages
  if (blueprint.restoreSources.length > 0) {
    nodes.push(makeRestoreNode(blueprint, nodes.length, emittedByOriginalIdPreRestore));
  }

  // 5. Emit post-restore stages
  const emittedByOriginalIdAllStages = new Map(emittedByOriginalIdPreRestore);
  for (let i = preCount; i < preCount + postCount; i += 1) {
    const stage = allStages[i];
    const ensuredStage: StageDraft = { ...stage, id: dedupedStageIds[i] };
    const stageNode = compileStageNode(ensuredStage, nodes.length, SENTINEL_MODEL_IDS);
    nodes.push(stageNode);
    emittedByOriginalIdAllStages.set(stage.id, {
      nodeId: stageNode.id,
      namedOutputIds: stage.namedOutputs.map((o) => o.id),
    });
  }

  // 6. node_export — may reference pre OR post stages
  nodes.push(makeExportNode(blueprint, nodes.length, emittedByOriginalIdAllStages));

  // 7. Linear edges
  const edges: WorkflowEdgeDef[] = [];
  for (let i = 0; i < nodes.length - 1; i += 1) {
    edges.push({
      id: `edge-${nodes[i].id}-${nodes[i + 1].id}`,
      source: nodes[i].id,
      target: nodes[i + 1].id,
    });
  }

  return {
    nodes,
    edges,
    sourceBlueprintHash: hashBlueprint(blueprint),
  };
}

function assertBlueprintCompilable(blueprint: FinalBlueprint): void {
  if (!Array.isArray(blueprint.inputFields) || blueprint.inputFields.length === 0) {
    throw new Error("COMPILE_BLUEPRINT: blueprint.inputFields is empty — no input fields to compile");
  }
  if (
    !blueprint.exportPlan ||
    !Array.isArray(blueprint.exportPlan.formats) ||
    blueprint.exportPlan.formats.length === 0
  ) {
    throw new Error("COMPILE_BLUEPRINT: blueprint.exportPlan.formats is empty");
  }
}

function buildStageOutputId(stageNodeId: string, modelId: string, namedOutputId: string): string {
  return `${stageNodeId}-model-${modelId}-artifact-${namedOutputId}`;
}

function makeInputNode(blueprint: FinalBlueprint, index: number): WorkflowNodeDef {
  const config: InputTransformConfig = {
    type: "input_transform",
    formFields: blueprint.inputFields,

  return {

    type: "input_transform",
    label: "输入转换",
    position: { x: index * X_STEP, y: 0 },

    outputs: deriveInputTransformOutputs("node_input", blueprint.inputFields),

}

function makeDesensNode(index: number): WorkflowNodeDef {
  const inputSources: InputSource[] = [
    {
      sourceNodeId: "node_input",
      outputId: "node_input-file-upload",
      displayName: "输入合并文本",
    },
  ];
  const config: DesensitizeConfig = {
    type: "desensitize",
    categories: [...DESENSITIZE_PLACEHOLDER_CATEGORIES],

    inputSources,

  return {

    type: "desensitize",
    label: "信息脱敏",
    position: { x: index * X_STEP, y: 0 },
    config,
    outputs: deriveDesensitizeOutputs("node_desens", inputSources),
  };
}

function makeRestoreNode(
  blueprint: FinalBlueprint,
  index: number,
  emittedByOriginalIdPreRestore: Map<string, StageLookupEntry>,
): WorkflowNodeDef {
  const inputSources: InputSource[] = blueprint.restoreSources.map((src) => {
    const entry = emittedByOriginalIdPreRestore.get(src.sourceStageId);
    if (!entry) {
      throw new Error(
        `COMPILE_BLUEPRINT: restore source references unknown or post-restore stage '${src.sourceStageId}'`,
      );
    }
    if (!entry.namedOutputIds.includes(src.sourceOutputId)) {
      throw new Error(
        `COMPILE_BLUEPRINT: restore source references unknown namedOutput '${src.sourceOutputId}' on stage '${src.sourceStageId}'`,
      );
    }
    return {
      sourceNodeId: entry.nodeId,
      outputId: buildStageOutputId(entry.nodeId, SENTINEL_MODEL_ID, src.sourceOutputId),
      displayName: src.label,
    };
  });

  const config: RestoreConfig = {
    type: "restore",
    pairedDesensitizeNodeId: blueprint.useDesensitize ? "node_desens" : null,
    inputSources,
  };

  return {
    id: "node_restore",
    type: "restore",
    label: "信息恢复",
    position: { x: index * X_STEP, y: 0 },
    config,
    outputs: deriveRestoreOutputs("node_restore", inputSources),
  };
}

function makeExportNode(
  blueprint: FinalBlueprint,
  index: number,
  emittedByOriginalIdAllStages: Map<string, StageLookupEntry>,
): WorkflowNodeDef {
  const contentMapping: VariableRef[] = blueprint.exportPlan.contentSources.map((cs) => {
    const entry = emittedByOriginalIdAllStages.get(cs.stageId);
    if (!entry) {
      throw new Error(
        `COMPILE_BLUEPRINT: export content source references unknown stage '${cs.stageId}'`,
      );
    }
    if (!entry.namedOutputIds.includes(cs.outputId)) {
      throw new Error(
        `COMPILE_BLUEPRINT: export content source references unknown namedOutput '${cs.outputId}' on stage '${cs.stageId}'`,
      );
    }
    return {
      nodeId: entry.nodeId,
      outputId: buildStageOutputId(entry.nodeId, SENTINEL_MODEL_ID, cs.outputId),
    };
  });

  const config: ExportConfig = {
    type: "export",
    formats: [...blueprint.exportPlan.formats],
    contentMapping,
  };

  return {
    id: "node_export",
    type: "export",
    label: "文件导出",
    position: { x: index * X_STEP, y: 0 },
    config,
    outputs: deriveExportOutputs(),
  };
}

function hashBlueprint(blueprint: FinalBlueprint): string {
  const json = JSON.stringify(blueprint, Object.keys(blueprint).sort());
  return createHash("sha256").update(json).digest("hex").slice(0, 8);
}

function deriveInputTransformOutputs(
  nodeId: string,
  formFields: FormFieldDef[],
): OutputDef[] {
  const outputs: OutputDef[] = [];
  for (const field of formFields) {
    if (field.type === "file") {
      if (field.fileSlotId) {
        outputs.push({
          id: `${nodeId}-fileslot-${field.fileSlotId}`,
          name: field.fileSlotLabel || field.label || "文件槽位",
          description: `文件槽位: ${field.fileSlotLabel || field.label}`,
          segmentKey: field.fileSlotId,
        });
      }
    } else {
      const key = field.machineKey || field.id;
      outputs.push({
        id: `${nodeId}-field-${key}`,
        name: field.label || "未命名",
        description: `用户输入项: ${field.label}`,
        segmentKey: key,
        category: "field",
      });
    }
  }
  const hasFileField = formFields.some((f) => f.type === "file");
  if (hasFileField) {
    outputs.push({
      id: `${nodeId}-file-upload`,
      name: "文件输出 (合并)",
      description: "所有文件合并文本",
      segmentKey: "text",
      category: "file_slot",
    });
  }
  return outputs;
}

function deriveDesensitizeOutputs(nodeId: string, inputSources: InputSource[]): OutputDef[] {
  if (inputSources.length > 0) {
    return inputSources.map((src) => ({
      id: `${nodeId}-desensitized-${src.outputId}`,
      name: `${src.displayName}.脱敏`,
      description: `脱敏后文本: ${src.displayName}`,
      segmentKey: src.outputId,
      category: "desensitized",
    }));
  }
  return [
    {
      id: `${nodeId}-desensitized`,
      name: "脱敏后文本",
      segmentKey: "desensitized",
      category: "desensitized",
    },
  ];
}

function deriveRestoreOutputs(nodeId: string, inputSources: InputSource[]): OutputDef[] {
  if (inputSources.length > 0) {
    return inputSources.map((src) => ({
      id: `${nodeId}-restored-${src.sourceNodeId}-${src.outputId}`,
      name: `${src.displayName}.恢复`,
      description: `恢复后文本: ${src.displayName}`,
      segmentKey: `${src.sourceNodeId}.${src.outputId}`,
      category: "restored",
    }));
  }
  return [
    {
      id: `${nodeId}-restored`,
      name: "恢复后文本",
      segmentKey: "restored",
      category: "restored",
    },
  ];
}

  return [];
