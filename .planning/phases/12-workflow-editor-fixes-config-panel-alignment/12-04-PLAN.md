```typescript
import { createSignal } from "solid-js";

export interface UndoRedoState<T> {
  current: () => T;
  push: (state: T) => void;
  undo: () => T | null;
  redo: () => T | null;
  canUndo: () => boolean;
  canRedo: () => boolean;
  reset: (state: T) => void;
}

export function createUndoRedo<T>(initial: T, maxHistory = 50): UndoRedoState<T> {
  const [undoStack, setUndoStack] = createSignal<T[]>([]);
  const [redoStack, setRedoStack] = createSignal<T[]>([]);
  const [current, setCurrent] = createSignal<T>(initial);

  function push(state: T) {
    setUndoStack((prev) => [...prev.slice(-(maxHistory - 1)), current()]);
    setRedoStack([]);
    setCurrent(() => state);
  }

  function undo(): T | null {
    const stack = undoStack();
    if (stack.length === 0) return null;
    const prev = stack[stack.length - 1];
    setUndoStack(stack.slice(0, -1));
    setRedoStack((r) => [...r, current()]);
    setCurrent(() => prev);
    return prev;
  }

  function redo(): T | null {
    const stack = redoStack();
    if (stack.length === 0) return null;
    const next = stack[stack.length - 1];
    setRedoStack(stack.slice(0, -1));
    setUndoStack((u) => [...u, current()]);
    setCurrent(() => next);
    return next;
  }

  function reset(state: T) {
    setUndoStack([]);
    setRedoStack([]);
    setCurrent(() => state);
  }

  return {
    current,
    push,
    undo,
    redo,
    canUndo: () => undoStack().length > 0,
    canRedo: () => redoStack().length > 0,
    reset,
  };
}
```

Major refactor of WorkflowEditor.tsx:

**1. Undo/redo integration:**
- Import `createUndoRedo` from hooks
- Create undo/redo instance: `const history = createUndoRedo<{ nodes: WFNode[]; edges: WFEdge[] }>({ nodes: [], edges: [] })`
- After loading workflow in onMount, call `history.reset({ nodes: sfNodes, edges: sfEdges })`
- Create a `pushSnapshot()` helper that serializes current `{ nodes: [...nodes], edges: [...edges] }` and calls `history.push()`. Call this BEFORE every state mutation (node add/delete, edge add/delete, config change, node move end — NOT on every drag pixel)
- Add keydown listener on the editor root div:
  - Ctrl+Z (no Shift): call `history.undo()`, if result is not null, apply to setNodes/setEdges
  - Ctrl+Shift+Z: call `history.redo()`, if result is not null, apply to setNodes/setEdges
- Show undo/redo buttons in toolbar (optional, keyboard is primary)

**2. Debounced autosave:**
- Add signals: `saveStatus: "idle" | "saving" | "saved" | "error"`, `lastSavedAt: string | null`
- Create a `handleAutoSave` function that does the PUT request (reuse existing save logic but WITHOUT the validate step)
- Use a debounce pattern: after any nodes/edges change, start a 1500ms timer. If another change comes in, restart timer. When timer fires, call `handleAutoSave`.
- Implementation: use `createEffect` watching a serialized key (e.g., `JSON.stringify({ nodes: nodes.map(n => ({id:n.id, position:n.position, data:n.data})), edges: edges.map(e => ({id:e.id, source:e.source, target:e.target})) })`) — on change, clear existing timeout, set new 1500ms timeout
- During save: set `saveStatus("saving")`; on success: set `saveStatus("saved")` and `lastSavedAt` to current time string; on error: set `saveStatus("error")`
- Use AbortController: if a new save starts while one is in-flight, abort the previous request

**3. Separate validation button:**
- Rename the existing "保存" button to "验证" (validate) button
- Add signal `validationStatus: "unvalidated" | "validated" | "invalid"`
- On validate button click: call the validate API (POST /validate), update validationErrors and validationStatus
- After any change (nodes/edges modified), reset `validationStatus` to `"unvalidated"`
- Keep the existing validation error overlay behavior

**4. Toolbar redesign:**
- Left side: back button, workflow name input (unchanged)
- Right side (in order):
  1. Autosave status indicator: small text showing "保存中..." (with spinner animation) | "已保存 HH:MM" | "保存失败" (red)
  2. Validation status badge: "已验证" (green badge) | "未验证" (gray badge) | validation error count button (red, same as current)
  3. Validate button: "验证" with check icon — separate from save
- Remove the manual "保存" button entirely (autosave replaces it)

**5. Undo/redo on autosave:**
- Per CONTEXT.md decision: undo/redo triggers autosave too. The debounce effect watches all state changes including those from undo/redo, so this happens automatically. No special handling needed.
