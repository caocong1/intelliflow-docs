1. In `packages/shared/src/types.ts`, add workflow-related types:

```typescript
/** The 5 workflow node types */
export type WorkflowNodeType = "input_transform" | "desensitize" | "model_call" | "restore" | "export";

/** Workflow status */
export type WorkflowStatus = "draft" | "active" | "disabled";

/** Named output (content block) from a node */
export interface OutputDef {
  id: string;
  name: string;
  description?: string;
}

/** Variable reference in prompt templates */
export interface VariableRef {
  nodeId: string;
  outputId: string;
  variableName: string;
}

/** Form field definition for input transform node */
export interface FormFieldDef {
  id: string;
  name: string;
  label: string;
  type: "text" | "textarea" | "file";
  required: boolean;
}

/** Node config discriminated union */
export interface InputTransformConfig {
  type: "input_transform";
  formFields: FormFieldDef[];
  allowFileUpload: boolean;
  acceptedFileTypes?: string[];
}

export interface DesensitizeConfig {
  type: "desensitize";
  ruleTypes: string[];
  placeholderFormat: string;
  localModelId: string | null;
}

export interface ModelCallConfig {
  type: "model_call";
  displayName: string;
  modelId: string | null;
  promptTemplate: string;
  inputRefs: VariableRef[];
}

export interface RestoreConfig {
  type: "restore";
  pairedDesensitizeNodeId: string | null;
}

export interface ExportConfig {
  type: "export";
  format: "word" | "pdf" | "markdown";
  templateId: string | null;
  contentMapping: VariableRef[];
}

export type NodeConfig = InputTransformConfig | DesensitizeConfig | ModelCallConfig | RestoreConfig | ExportConfig;

/** A node instance in a workflow */
export interface WorkflowNodeDef {
  id: string;
  type: WorkflowNodeType;
  label: string;
  position: { x: number; y: number };
  config: NodeConfig;
  outputs: OutputDef[];
}

/** An edge connecting two nodes */
export interface WorkflowEdgeDef {
  id: string;
  source: string;
  target: string;
  sourceHandle?: string;
  targetHandle?: string;
}

/** Validation error from workflow validation */
export interface WorkflowValidationError {
  nodeId?: string;
  field?: string;
  message: string;
  severity: "error" | "warning";
}

/** Workflow entity */
export interface Workflow extends BaseEntity {
  documentTypeId: string;
  name: string;
  description: string | null;
  status: WorkflowStatus;
  isDefault: boolean;
  nodes: WorkflowNodeDef[];
  edges: WorkflowEdgeDef[];
  schemaVersion: number;
}

/** Workflow list item (without full graph data for list pages) */
export interface WorkflowListItem extends BaseEntity {
  documentTypeId: string;
  documentTypeName: string;
  name: string;
  description: string | null;
  status: WorkflowStatus;
  isDefault: boolean;
  nodeCount: number;
}
```

<task type="auto">
  <name>Task 2: Backend workflow CRUD API + validation engine</name>
  <files>packages/backend/src/modules/workflows/workflows.routes.ts, packages/backend/src/modules/workflows/workflows.service.ts, packages/backend/src/modules/workflows/validation.ts, packages/backend/src/index.ts</files>
  <action>
1. Create `packages/backend/src/modules/workflows/validation.ts`:
   - `validateWorkflow(nodes: WorkflowNodeDef[], edges: WorkflowEdgeDef[]): WorkflowValidationError[]`
   - Rules:
     - Must have at least one `input_transform` node
     - Must have at least one `export` node
     - Every `desensitize` node must have a `restore` node reachable downstream (follow edges)
     - Every `restore` node must have `pairedDesensitizeNodeId` set and pointing to an existing desensitize node
     - No orphan nodes (every node connected by at least one edge, except if only 1 node)
     - No cycles (topological sort — if fails, cycle exists)
     - Required fields per node type: model_call needs `promptTemplate`, export needs `format`
     - Referenced `modelId` validation deferred (just check non-null if type is model_call)

2. Create `packages/backend/src/modules/workflows/workflows.service.ts`:
   - `listWorkflows(db, { documentTypeId?, search?, page, pageSize })` — paginated list with JOIN to documentTypes for name, returns WorkflowListItem[] (compute nodeCount from JSONB array length using `sql`). Do NOT return full nodes/edges in list.
   - `getWorkflow(db, id)` — full workflow with nodes/edges
   - `createWorkflow(db, { documentTypeId, name, description })` — creates with status "draft", empty nodes/edges
   - `updateWorkflow(db, id, { name?, description?, nodes?, edges? })` — updates fields, sets updatedAt
   - `deleteWorkflow(db, id)` — hard delete
   - `toggleWorkflowStatus(db, id, status)` — if enabling to "active", run validateWorkflow first; reject if errors exist. If current status is "active" and new is "disabled", allow directly.
   - `copyWorkflow(db, id, { name, targetDocumentTypeId? })` — deep copy with new ID, status "draft", isDefault false
   - `setDefaultWorkflow(db, id)` — in a transaction: unset isDefault for all workflows of same documentTypeId, then set this one. Only active workflows can be set as default.

3. Create `packages/backend/src/modules/workflows/workflows.routes.ts`:
   - Follow existing pattern: `new Elysia({ prefix: "/workflows" })` with authPlugin + requireAdmin
   - Routes:
     - `GET /` — list with query params (documentTypeId, search, page, pageSize)
     - `GET /:id` — get single workflow with full graph
     - `POST /` — create (body: documentTypeId, name, description)
     - `PUT /:id` — update (body: name?, description?, nodes?, edges?)
     - `DELETE /:id` — delete
     - `POST /:id/validate` — validate and return errors array
     - `POST /:id/copy` — copy (body: name, targetDocumentTypeId?)
     - `PATCH /:id/status` — toggle status (body: status)
     - `PATCH /:id/set-default` — set as default for document type

4. Register in `packages/backend/src/index.ts`: import and `.use(workflowRoutes)` after modelRoutes.
