  // Helper: add node

  // Helper: remove node + associated edges

  // Helper: remove nodes (batch) + associated edges
  function removeNodes(ids: Set<string>) { ... }
  // Helper: add edge (enforce linear: remove existing edges from source or to target)
  function addEdge(edge: FlowEdgeData) { ... }
  // Helper: remove edge

  // Helper: remove edges (batch)

  // Helper: get snapshot for undo/redo and autosave
  function getSnapshot(): { nodes: FlowNodeData[]; edges: FlowEdgeData[] } { ... }
  // Helper: apply snapshot (from undo/redo)

Implement all helpers.
