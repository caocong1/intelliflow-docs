## 1a. Add InputSource type and update configs (types.ts)

Add shared InputSource interface:
```typescript
/** Explicit input source reference for desensitize/restore nodes */
export interface InputSource {
  sourceNodeId: string;
  outputId: string;
  displayName: string;  // auto-filled from upstream output name
}
```

Add `inputSources: InputSource[]` to `DesensitizeConfig` and `RestoreConfig`.

## 1b. Update derive-outputs.ts

Change desensitize case to generate one output per inputSource:
```typescript
case "desensitize":
  if (config.inputSources && config.inputSources.length > 0) {
    return config.inputSources.map((src) => ({
      id: `${nodeId}-desensitized-${src.outputId}`,
      name: `${src.displayName}.脱敏`,
      description: `脱敏后文本: ${src.displayName}`,
    }));
  }
  // Fallback for legacy configs without inputSources
  return [{ id: `${nodeId}-desensitized`, name: "脱敏后文本" }];
```

Same pattern for restore:
```typescript
case "restore":
  if (config.inputSources && config.inputSources.length > 0) {
    return config.inputSources.map((src) => ({
      id: `${nodeId}-restored-${src.outputId}`,
      name: `${src.displayName}.恢复`,
      description: `恢复后文本: ${src.displayName}`,
    }));
  }
  return [{ id: `${nodeId}-restored`, name: "恢复后文本" }];
```
